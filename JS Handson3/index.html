Q3.Find Sum

        const findSum = (n) => {
            var sumEven = 0;
            for(i=2;i<=n;i++)
            {
              if(i%2===0)
              {
                sumEven = sumEven + i;

Q6.Print the Pattern.

        const Print_pattern = (N) => 
        {
          var pattern = "";
          for(let i=1;i<=N;i++)
          {
            for(let j=1;j<=i;j++)
            {
              pattern = pattern + "*";
            }
            pattern = pattern + "\n";
          }
          console.log(pattern);
        };

Q7.Check whether a Number is a prime or not.

        const Prime_Check = (N) => 
        {
          if(N==1)
          {
            return "NO";
          }
          for(i=2;i<=Math.sqrt(N);i++)
          {
            if(N%i===0)
            {
              return "NO";
            }
          }
          return "YES";
        };
